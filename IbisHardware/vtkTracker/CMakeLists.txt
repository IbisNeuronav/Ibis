SET ( Local_SRCS
vtkTracker.cxx

vtkTrackerBuffer.cxx

)


SET ( NDI_SRCS  ndicapi.c
                ndicapi_serial.c
                ndicapi_thread.c

               ndicapi_serial.h
               ndicapi_thread.h
               ndicapi_math.h )

                   vtkTrackerTool.cxx
                   vtkTrackerBuffer.cxx
                   vtkPOLARISTracker.cxx
                   vtkFrameToTimeConverter.cxx
                   ndicapi.c
                   ndicapi_serial.c
                   ndicapi_thread.c
                   ndicapi_math.c )

                  vtkTrackerBuffer.h
                  vtkPOLARISTracker.h
                  vtkFrameToTimeConverter.h

                  ndicapi_serial.h
                  ndicapi_thread.h
                  ndicapi_math.h )

vtkTracker.cxx
vtkAbstractFunction.cxx

)
